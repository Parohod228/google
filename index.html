let link = null; // тепер дефолт — null window.onload = function() { if (link === null) { sessionStorage.setItem("IW", "false"); console.log("No redirect, link is null."); return; // нічого не робимо } else { sessionStorage.setItem("IW", "true"); sessionStorage.setItem("L", link); } const current = location.href; const parohod = "https://parohod228.github.io/google"; if (current.startsWith(link) || current.startsWith(parohod)) { console.log("Already on target site, no redirect."); return; } console.log("Redirecting to:", link); window.location.replace(link); };
